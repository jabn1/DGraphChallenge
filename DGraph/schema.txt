type Product {
    id: String! @id
    name: String!
    price: String!
    date: DateTime!
}

type Buyer {
    id: String! @id
    name: String!
    age: Int!
    date: DateTime!
}

type Transaction {
    id: String! @id
    buyer_id: String! @search(by: [exact])
    ip: String! @search(by: [exact])
    device: String!
    product_ids: [String]! 
    date: DateTime!
}
